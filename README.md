import copy

def get_user(user_id):
    """
    Simula o GET /users/{id}
    Busca um usuário na base fictícia/fake.
    """
    user = fake_db.get(user_id)
    return copy.deepcopy(user) if user else None


def generate_ai_news(user):
    """
    Simula a etapa de transformação.
    Em vez de chamar a OpenAI, gera uma mensagem baseada no saldo.
    """
    nome = user["name"]
    saldo = user["account"]["balance"]

    if saldo < 500:
        mensagem = f"{nome}, comece a investir aos poucos e fortaleça seu futuro."
    elif saldo < 2000:
        mensagem = f"{nome}, diversificar investimentos pode proteger seu patrimônio."
    else:
        mensagem = f"{nome}, seu perfil favorece estratégias para ampliar seus ganhos."

    return mensagem


def update_user(user):
    """
    Simula o PUT /users/{id}
    Atualiza o usuário na base fake.
    """
    user_id = user["id"]

    if user_id in fake_db:
        fake_db[user_id] = copy.deepcopy(user)
        return True
    return False

import requests
import pandas as pd
from openai import RateLimitError

# URL da sua API FastAPI exposta
api_url = "INSERIR_API_URL"

def api_get_user(user_id):
    """
    Busca um usuário na API pelo ID.
    Equivale ao GET /users/{id}
    """
    response = requests.get(f"{api_url}/users/{user_id}")

    if response.status_code == 200:
        return response.json()
    else:
        print(f"Erro ao buscar usuário {user_id}: {response.status_code}")
        return None

def api_update_user(user):
    """
    Atualiza um usuário na API.
    Equivale ao PUT /users/{id}
    """
    response = requests.put(f"{api_url}/users/{user['id']}", json=user)
    return response.status_code == 200
